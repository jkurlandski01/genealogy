Pass a method to another method as a parameter.

	// JERRY: angle brackets need to be escaped
// Method definition.
    /**
     * If input pos is a noun, return a higher score than if it isn't a noun.
     */
    public static final Function scoreNounsHigher = (pos) -> {
        EnumSet bestPos = EnumSet.of(PennEnPos.NN, PennEnPos.NNP, PennEnPos.NNPS, PennEnPos.NNS);
        double score = bestPos.contains(pos) ? 0.5 : 0.25;
        return score;
    };


// Method invocation.
    /**
     * Find each node in the tree that is of the given type, and assign to it the given confidence score.
     */
    public static Map scoreAllNodesOfType(IDepParseTree tree, ITaxonomy type, Function func) {
        Map result = Maps.newHashMap();

        tree.getDepNodes().stream().filter(node -> type.isType(node.lemma())).forEach(node -> {
            PennEnPos pos = node.pos();
            double score = func.apply(pos);
            result.put(node, score);
        });

        return result;
    }

// Passing the method as a parameter.
result = Taxonomy.scoreAllNodesOfType(sentence, type, Taxonomy.scoreNounsHigher);
